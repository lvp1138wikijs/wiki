Following query is used to limit cpu of vm from whatever is assigned to it as per package to below down to 50%

INSERT INTO sc_trigger (server,sc_trigger,domain,param1,param2) values('vm','vmcpucap','username','vm_id','50');

Eg:- 

INSERT INTO sc_trigger (server,sc_trigger,domain,param1,param2) values('vm','vmcpucap','michae44','2b42f0735d6f021cb3ab3da0dd7a781c','50');

 

This query can be used without username too, just leave user name empty

INSERT INTO sc_trigger (server,sc_trigger,domain,param1,param2) values('vm','vmcpucap','','vm_id','50');

Eg: -

INSERT INTO sc_trigger (server,sc_trigger,domain,param1,param2) values('vm','vmcpucap','','2b42f0735d6f021cb3ab3da0dd7a781c','50');
