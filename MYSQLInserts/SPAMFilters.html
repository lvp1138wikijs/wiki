Custom Score

To set custom score for a domain

DELETE FROM email_properties WHERE sc_domain = "_domain_" AND username = "_username_";

INSERT INTO email_properties(username,sc_domain,sc_feature,sc_value) values ('_username_','_domain-name_','score','16');

To  reset custom score for a domain to default

DELETE FROM email_properties where username='_username_' and sc_domain='_domain-name_' and sc_feature='score';

To set custom score for an email account (i.e postmaster@jftechnologies.net)

INSERT INTO email_properties(username,sc_emailuser,sc_domain,sc_feature,sc_value) values ('_username_','_email-user_','_domain-name_','score','16');

To  reset custom score for an email account

DELETE FROM email_properties where username='_username_' and sc_emailuser='_email-user_' and  sc_domain='_domain-name_' and sc_feature='score';

 

 

Enable/Disable Complete Spam Checking

 

To Disable spam checking for a domain (Full spam checking including smtpd, simscan and spamassassin)

INSERT INTO email_properties(username,sc_domain,sc_feature,sc_value) values ('_username_','_domain-name_','av','0');

To reset and again enable  spam checking for a domain (Full spam checking including smtpd, simscan and spamassassin)

DELETE FROM email_properties where username='_username_' and sc_domain='_domain-name_' and sc_feature='av';

INSERT INTO email_properties(username,sc_domain,sc_feature,sc_value) values ('_username_','_domain-name_','av','1');

To Disable spam checking for an email account (i.e postmaster@jftechnologies.net)

INSERT INTO email_properties(username,sc_emailuser,sc_domain,sc_feature,sc_value) values ('_username_','_email-user_','_domain-name_','av','0');

 

To reset and again enable  spam checking for a domain (Full spam checking including smtpd, simscan and spamassassin)

DELETE FROM email_properties where username='_username_' and sc_emailuser='_email-user_' and  sc_domain='_domain-name_' and sc_feature='av';

INSERT INTO email_properties(username,sc_emailuser,sc_domain,sc_feature,sc_value) values ('_username_','_email-user_','_domain-name_','av','1');

 

Enable/Disable Reverse DNS

To Disable rdns checking for a domain

INSERT INTO email_properties(username,sc_domain,sc_feature,sc_value) values ('_username_','_domain-name_','check_ip_reverse','0');

To reset and again enable  rdns checking for a domain

DELETE FROM email_properties where username='_username_' and sc_domain='_domain-name_' and sc_feature='check_ip_reverse';

INSERT INTO email_properties(username,sc_domain,sc_feature,sc_value) values ('_username_','_domain-name_','check_ip_reverse','1');
