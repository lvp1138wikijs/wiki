Secure.Serverpoint

Enable PHP

rpm -Uvh http://rpms.famillecollet.com/enterprise/remi-release-7.rpm

To Enable PHP5 : 

yum-config-manager --enable remi-php56

To Enable PHP7

yum-config-manager --enable remi-php73 ; yum-config-manager --enable remi-php74

 

Install Packages 

yum install epel-release yum-utils -y

yum install -y httpd mod_ssl mariadb php php-cli mod_fcgid php-cgi php-mysql php-common \
php-gd php-mcrypt php-imap php-ldap iptables-services perl-DBD-SQLite perl-XML-Smart perl-URI-Encode -y

 

Configure Apache & PHP

service httpd start ; systemctl enable httpd.service
rm -rf /var/www/cgi-bin
groupadd webroot
mkdir -p /home/users/s/
useradd -s /bin/bash -d /home/users/s/serverpoint.com -m -g webroot webroot
passwd webroot
mkdir -p /home/users/s/serverpoint.com/public_html

sed -i 's/;cgi.fix_pathinfo=1/cgi.fix_pathinfo=1/g' /etc/php.ini
sed -i 's/short_open_tag = Off/short_open_tag = On/g' /etc/php.ini

 

cat >> /home/users/s/serverpoint.com/public_html/cgi-bin/php.fcgi <<END
#!/bin/bash
PHP_CGI=/usr/bin/php-cgi
PHP_FCGI_CHILDREN=4
PHP_FCGI_MAX_REQUESTS=0
export PHP_FCGI_CHILDREN
export PHP_FCGI_MAX_REQUESTS
exec $PHP_CGI
END

 

chmod 755 /home/users/s/serverpoint.com/public_html/cgi-bin/php.fcgi
chown -R webroot:webroot /home/users/s/serverpoint.com/public_html
